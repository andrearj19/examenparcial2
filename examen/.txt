Perfecto — corrijo el código roto y lo convierto en totalmente funcional. Mantendré un estilo funcional y seguro:

filtra por páginas > minPaginas y año >= minAnio,

calcula correctamente por autores (sin duplicados),

maneja el caso "sin resultados" devolviendo Option[AutorInfo],

case class Libro(titulo: String, autor: String, paginas: Int, anio: Int)
case class AutorInfo(autor: String, totalPaginas: Int, cantidadLibros: Int)

def autorMasProductivoCorregido(
  libros: List[Libro],
  minPaginas: Int,
  minAnio: Int
): Option[AutorInfo] = {

  // 1) Filtrar por páginas (>) y año (>=)
  val filtrados = libros.filter(l => l.paginas > minPaginas && l.anio >= minAnio)

  // Si no hay libros que cumplan las condiciones, devolver None
  if (filtrados.isEmpty) return None

  // 2) Obtener autores únicos a partir de los libros filtrados
  val autoresUnicos: List[String] = filtrados.map(_.autor).distinct

  // 3) Para cada autor, calcular total de páginas y cantidad de libros (solo sobre 'filtrados')
  val infoPorAutor: List[AutorInfo] = autoresUnicos.map { autor =>
    val librosDelAutor = filtrados.filter(_.autor == autor)
    val totalPaginas = librosDelAutor.map(_.paginas).sum
    val cantidadLibros = librosDelAutor.length
    AutorInfo(autor, totalPaginas, cantidadLibros)
  }

  // 4) Seleccionar al autor con mayor totalPaginas de forma segura
  if (infoPorAutor.isEmpty) None
  else Some(infoPorAutor.maxBy(_.totalPaginas))
}
